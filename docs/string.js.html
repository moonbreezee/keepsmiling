/**
 * @function
 * @description 检测字符串是不是回文结构
 * @param {String} str - 需要检测的字符串
 * @return {Boolean} 返回的boolan值
 * @example
 * isPalindrome("level");
 * // returns true
 * isPalindrome("levels");
 * // returns false
 */
export function stringIsPalindrome(str) {
  str = str.replace(/W/g, "").toLowerCase();
  return (str === str.split("").reverse().join(""));
}
/**
 * 大写每个单词的首字母
 * @summary 使用replace（）匹配每个单词的第一个字符
 * @function
 * @param {String} str - 需要判断的字符串
 * @return {Boolean}
 * @example
 * stringCapitalizeEveryWord('hello world!')
 * // 'Hello World!'
 */
export const stringCapitalizeEveryWord = str => str.replace(/\b[a-z]/g, char => char.toUpperCase());
